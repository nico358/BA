384 uint8_t adapter_find_sensors(uint8_t *addresses, struct BUS_INTERFACE *bus_interface, delay_function_ptr delay_fptr)
385 {
386  uint8_t no_match = 1;
387  uint8_t count = 0;
388  uint8_t res = 0;
389  /* Loop over all possible sensor addresses */
390  for (uint8_t loop_var = 0; loop_var < SENSOR_ADDRESS_NUMBER; loop_var++)
391  {
392  /* Loop for max_search_attempts times and poll sensor address */
393  while (no_match && count <= max_search_attempts)
394  {
395  /* Call address by bus communication and set address in array when match */
396  no_match = poll_fbus(addresses, bus_interface, loop_var);
397  /* Increment brak condition and sleep 10ms */
398  if (no_match)
399  {
400  count++;
401  delay_fptr(10);
402  }
403  }
404  /* If sensor has responsed count match results up */
405  if(!no_match)
406  {
407  res++;
408  }
409  /* reset break conditions */
410  no_match = 1;
411  count = 0;
412  }
413  return res;
414 }

419 int8_t adapter_fbus_write(const uint8_t sensor_address, const uint8_t reg_address, uint8_t *data_ptr, const uint16_t len)
420 {
421  uint8_t result = PAC1720_OK;
422  /* Send start communication (blocking) */
423  internal_bus_interface->startWait((sensor_address << BUS_ADDRESS_SHIFT) + I2C_WRITE);
424  /* Write register address */
425  result = internal_bus_interface->write(reg_address);
426  if(result != PAC1720_OK)
427  {
428  return result;
429  }
430  /* Loop over data and send */
431  result = fbus_write_loop(data_ptr, len);
432 
434  internal_bus_interface->stop();
435  return result;

492 int8_t fbus_read_loop(uint8_t *data_ptr, const uint16_t len)
493 {
494  uint8_t result = PAC1720_OK;
495  /* Read from #len registers, increase the address #len times */
496  for(uint16_t i = 0; i < len; i++)
497  {
498  if(i < len -1) *data_ptr = internal_bus_interface->readAck();
499  /* After the last read send NACK */
500  else *data_ptr = internal_bus_interface->readNak();
501  data_ptr++;
502  }
503  return result;
504 }

509 uint8_t poll_fbus(uint8_t *addresses, struct BUS_INTERFACE *bus_interface, uint8_t loop_var)
510 {
511  /* Get the current sensor address to poll out of lookup-table, indexed by loop variable */
512  uint8_t sensor_addr = PAC1720_addresses[loop_var];
513  /* Start communication (non-blocking) and stop afterwards*/
514  uint8_t no_match = bus_interface->start((sensor_addr << BUS_ADDRESS_SHIFT) + I2C_WRITE);
515  bus_interface->stop();
516  /* If sensor has ACKed store address at its index in the array */
517  if(!no_match)
518  {
519  uint8_t *offset = addresses + loop_var;
520  *offset = sensor_addr;
521  }
522  /* Return match or fail */
523  return no_match;
524 }
