85  def processFileByLine(self):
86  """TODO."""
87  if not self.filepath is None:
88  # Assemble measurement location
89  path = self.folderpath + self.filepath + '.txt'
90  try:
91  # Readin of multiple sensor measurements from one file not supported
92  with open (path, "r") as reader:
93  if reader.mode == 'r':
94  # Set matadata and measurement header
95  self.appendHead()
96  reader.readline()
97  line = reader.readline()
98  while(line):
99  # Split incoming line
100  splinter = self.splitStr("{}".format(line.replace('\r\n', '')))
101  # Set metadata ch1 at first loop
102  if len(self.tmp_meas_ch1[LIST_INDEX_META]) == 1:
103  self.name_ch1 = splinter[CH1_NAME_INDEX]
104  self.setMetaCh1()
105  # Set metadata ch2 at first loop
106  if len(self.tmp_meas_ch2[LIST_INDEX_META]) == 1:
107  self.name_ch2 = splinter[CH2_NAME_INDEX]
108  self.setMetaCh2()
109  # Append measurements, must be 9 values, drop incomplete
110  if len(splinter) == 9:
111  self.setListValues(splinter)
112  # Read next incoming line
113  line = reader.readline()
114  # Set matadata and measurement footer
115  self.appendFoot()
116  except (ValueError, OSError, IOError) as e:
117  ExceptionHandler(e, "processFileByLine")
118  # Finally store formatted data
119  self.writeFormattedDataToTxtFile(self.tmp_meas_ch1, self.name_ch1)
120  self.writeFormattedDataToTxtFile(self.tmp_meas_ch2, self.name_ch2)
121  # Plot formatted data
122  self.plotFormattedDataToPngFile(self.overlay)
