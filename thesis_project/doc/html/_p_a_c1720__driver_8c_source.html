1590 float calculate_SENSED_VOLTAGE(const uint16_t *v_sense_voltage_reg_ptr, const uint8_t *current_sense_sampling_time_reg_ptr)
1591 {

1593  if(is_negative_value(v_sense_voltage_reg_ptr)){
1594  /* If number is negative build two's complement */
1595  uint16_t complement = twos_complement(v_sense_voltage_reg_ptr);
1596  /* Shift unused bits out of result according to resolution,
1597  shift is determined by lookuptable indexed by configured sampling time value */
1598  uint16_t tmp = right_bit_shift(&complement, CURRENT_RESOLUTION_IGNORE_BITS[*current_sense_sampling_time_reg_ptr]);
1599  /* Cast result to float */
1600  float res = (float) tmp;
1601  /* Sign result */
1602  return - res;
1603  } else {
